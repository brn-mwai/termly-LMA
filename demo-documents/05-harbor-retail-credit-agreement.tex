\section{DEFINITIONS}

\textbf{"Borrower"} means Harbor Retail Group, LLC, a California limited liability company operating retail stores throughout the Western United States.

\textbf{"Commitment Amount"} means Thirty-Five Million Dollars (\$35,000,000).

\textbf{"EBITDA"} means Net Income plus Interest Expense, taxes, depreciation, amortization, and non-cash charges, minus non-cash gains.

\textbf{"EBITDAR"} means EBITDA plus Rent Expense.

\textbf{"Fixed Charge Coverage Ratio"} means (EBITDA minus Capital Expenditures minus Taxes Paid) divided by (Interest Expense plus Scheduled Principal Payments).

\textbf{"Interest Coverage Ratio"} means EBITDA divided by Interest Expense.

\textbf{"Leverage Ratio"} means Total Debt divided by EBITDA.

\textbf{"Maturity Date"} means June 15, 2027.

\section{THE CREDIT FACILITY}

\subsection{Term Loan and Revolver}
\begin{itemize}
    \item Term Loan: \$25,000,000
    \item Revolving Credit Facility: \$10,000,000
\end{itemize}

\subsection{Interest Rate}
SOFR plus 4.00\% (increased from 3.50\% following Q2 2025 covenant breach waiver).

\subsection{Collateral}
All assets of the Borrower, including inventory, accounts receivable, equipment, and real property.

\section{FINANCIAL COVENANTS}

\subsection{Maximum Leverage Ratio}
The Borrower shall maintain a Leverage Ratio not exceeding \textbf{4.50:1.00}.

\subsection{Minimum Interest Coverage Ratio}
The Borrower shall maintain an Interest Coverage Ratio of not less than \textbf{2.00:1.00}.

\subsection{Minimum Fixed Charge Coverage Ratio}
The Borrower shall maintain a Fixed Charge Coverage Ratio of not less than \textbf{1.10:1.00}.

\subsection{Maximum Capital Expenditures}
Capital Expenditures shall not exceed \textbf{\$3,000,000} in any fiscal year without Lender approval.

\section{REPORTING REQUIREMENTS}

The Borrower shall deliver:
\begin{itemize}
    \item Monthly financial statements within 30 days of month-end
    \item Quarterly Compliance Certificate within 45 days of quarter-end
    \item 13-week cash flow forecast, updated bi-weekly
    \item Annual audited financials within 90 days of fiscal year-end
\end{itemize}

\section{EVENTS OF DEFAULT}

Events of Default include:
\begin{enumerate}
    \item Non-payment of principal or interest
    \item Breach of any financial covenant
    \item Material adverse change in business condition
    \item Cross-default on other indebtedness
    \item Change of control without consent
\end{enumerate}

\section{REMEDIES}

Upon an Event of Default, Lender may:
\begin{itemize}
    \item Accelerate all outstanding obligations
    \item Increase interest rate by 2.00\%
    \item Exercise remedies against collateral
    \item Require immediate repayment
\end{itemize}
